递归实现栈逆序

#include <iostream>
#include <deque>
#include <stack>

template<typename T>
void ReverseStack(std::stack<T>& stack) {
  if (!stack.empty()) {
    T top = stack.top();
    stack.pop();
    ReverseStack(stack);
    AddToStackBottom(stack, top);
  }
}

template<typename T>
void AddToStackBottom(std::stack<T>& stack, T value) {
  if (stack.empty()) {
    stack.push(value);
  } else {
    T top = stack.top();
    stack.pop();
    AddToStackBottom(stack, value);
    stack.push(top);
  }
}


template<typename T>
void Print(std::stack<T>& stack) {
  while (!stack.empty()) {
    T top = stack.top();
    std::cout << top << " ";
    stack.pop();
  }
  std::cout << std::endl;
}

int main() {
  std::deque<int> vv = {1,2,3,4,5};
  std::stack<int> stack(vv);
  //Print(stack);
  ReverseStack(stack);
  Print(stack);
}
