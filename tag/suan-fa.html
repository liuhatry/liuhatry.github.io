动态规划之最长公共子序列（不连续）

#include <iostream>
#include <string>

using std::string;

void LCSPrint(const string& x, const string& y, int (*direction)[100], size_t row, size_t col) {
    if (row <=0 || col <= 0)
        return;
    if (1 == direction[row][col]) {
        LCSPrint(x, y, direction, row-1, col-1);
        std::cout << x[row-1];
    } else if (2 == direction[row][col]) {
        LCSPrint(x, y, direction, row, col-1);
    } else if (3 == direction[row][col]) {
        LCSPrint(x, y, direction, row-1, col);
    }
}

bool LCS(const string& x, const string& y) {
    size_t len_x = x.length();
    size_t len_y = y.length();
    if (0 == len_x || 0 == len_y) {
        return false;
    }

    int LCS_length[100][100];
    int LCS_direction[100][100];
    for (size_t i = 0; i <= len_x; i++) {
        LCS_length[i][0] = 0;
    }
    for (size_t i = 0; i <= len_y; i++) {
        LCS_length[0][i] = 0;
    }

    for (size_t row = 1; row <= len_x; row++) {
        for (size_t col = 1; col <=len_y; col++) {
            if (x[row-1] == y[col-1]) {
                LCS_length[row][col] = LCS_length[row-1][col-1] + 1;
                LCS_direction[row][col] = 1;
            } else if (LCS_length[row][col-1] >= LCS_length[row-1][col]) {
                LCS_length[row][col] = LCS_length[row][col-1];
                LCS_direction[row][col] = 2;
            } else {
                LCS_length[row][col] = LCS_length[row-1][col];
                LCS_direction[row][col] = 3;
            }
        }
    }
    std::cout << "The length of LCS:" << std::endl;
    std::cout << LCS_length[len_x][len_y] << std::endl;
    std::cout << "The LCS:" << std::endl;
    LCSPrint(x, y, LCS_direction, len_x, len_y);
    std::cout << std::endl;
    return true;
}

int main(int argc, char *argv[]){
    if (3 != argc){
        std::cout << "Usage: " << argv[0] << " str1 str2" << std::endl;
        return -1;
    }
    string x = argv[1];
    string y = argv[2];
    LCS(x,y);
}
